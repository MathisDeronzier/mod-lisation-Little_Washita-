source(paste(loc,"/codes_utiles/test_stat_Cramer_von_Mises.R",sep=""))
## Warning: package 'goftest' was built under R version 4.0.3
#On applique l'algorithme de Cramer-von Mises
CVM_simu_deb<-function(serie){return(cvm(serie,simu_deb[,1],TRUE))}
CVM_simu_etpr<-function(serie){return(cvm(serie,simu_etpr[,1],TRUE))}
deb_CVM<-apply(simu_deb, 2, CVM_simu_deb)
etpr_CVM<-apply(simu_etpr, 2, CVM_simu_etpr)
#CVM$T # donne la valeur de la "distance" entre les cdfs
#CVM$pval # p-value du test CvM 
deb_CVM_stat<-matrix(0,1,5)
etpr_CVM_stat<-matrix(0,1,5)
for (i in (1:5)){
  deb_CVM_stat[i]<-deb_CVM[[i]]$T
  etpr_CVM_stat[i]<-etpr_CVM[[i]]$T
}
layout(matrix(1:2,1,2))
deg<-c(1,9,25,49,81)
plot(deg,deb_CVM_stat, main= "dist CvM s√©ries prec deg", xlab="deg" , ylab="CvM", type="l")
plot(deg,etpr_CVM_stat, main= "dist CvM s√©ries etp deg", xlab="deg" , ylab="CvM", type="l")

On voit, alors que la distance de CvM √©tait tr√®s grande pour les pr√©cipitations, elle est quasiment nulle pour les d√©bit. C‚Äôest un bon indicateur pour estimer la qualit√© de nos pr√©dictions.
